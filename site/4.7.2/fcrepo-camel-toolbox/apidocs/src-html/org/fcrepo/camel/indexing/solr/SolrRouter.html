018package org.fcrepo.camel.indexing.solr;
019
020import static java.util.stream.Collectors.toList;
021import static org.apache.camel.Exchange.CONTENT_TYPE;
022import static org.apache.camel.Exchange.HTTP_METHOD;
023import static org.apache.camel.Exchange.HTTP_QUERY;
024import static org.apache.camel.Exchange.HTTP_URI;
025import static org.apache.camel.builder.PredicateBuilder.and;
026import static org.apache.camel.builder.PredicateBuilder.in;
027import static org.apache.camel.builder.PredicateBuilder.not;
028import static org.apache.camel.builder.PredicateBuilder.or;
029import static org.fcrepo.camel.FcrepoHeaders.FCREPO_EVENT_TYPE;
030import static org.fcrepo.camel.FcrepoHeaders.FCREPO_RESOURCE_TYPE;
031import static org.fcrepo.camel.FcrepoHeaders.FCREPO_URI;
032import static org.fcrepo.camel.processor.ProcessorUtils.tokenizePropertyPlaceholder;
033import static org.slf4j.LoggerFactory.getLogger;
034
035import org.apache.camel.LoggingLevel;
036import org.apache.camel.builder.RouteBuilder;
037import org.apache.camel.builder.xml.Namespaces;
038import org.fcrepo.camel.processor.EventProcessor;
039import org.slf4j.Logger;
040
041/**
042 * A content router for handling JMS events.
043 *
044 * @author Aaron Coburn
045 */
046public class SolrRouter extends RouteBuilder {
047
048    private static final Logger logger = getLogger(SolrRouter.class);
049
050    private static final String hasIndexingTransformation =
051        "(/rdf:RDF/rdf:Description/indexing:hasIndexingTransformation/@rdf:resource | " +
052        "/rdf:RDF/rdf:Description/indexing:hasIndexingTransformation/@rdf:about)[1]";
053
054    private static final String RESOURCE_DELETION = "http://fedora.info/definitions/v4/event#ResourceDeletion";
055    private static final String DELETE = "https://www.w3.org/ns/activitystreams#Delete";
056    private static final String INDEXING_TRANSFORMATION = "CamelIndexingTransformation";
057    private static final String INDEXABLE = "http://fedora.info/definitions/v4/indexing#Indexable";
058    private static final String INDEXING_URI = "CamelIndexingUri";
059
060    /**
061     * Configure the message route workflow.
062     */
063    public void configure() throws Exception {
064
065        final Namespaces ns = new Namespaces("rdf", "http://www.w3.org/1999/02/22-rdf-syntax-ns#");
066        ns.add("indexing", "http://fedora.info/definitions/v4/indexing#");
067        ns.add("ldp", "http://www.w3.org/ns/ldp#");
068
069
070        /*
071         * A generic error handler (specific to this RouteBuilder)
072         */
073        onException(Exception.class)
074            .maximumRedeliveries("{{error.maxRedeliveries}}")
075            .log("Index Routing Error: ${routeId}");
076
077        /*
078         * route a message to the proper queue, based on whether
079         * it is a DELETE or UPDATE operation.
080         */
081        from("{{input.stream}}")
082            .routeId("FcrepoSolrRouter")
083            .process(new EventProcessor())
084            .choice()
085                .when(or(header(FCREPO_EVENT_TYPE).contains(RESOURCE_DELETION),
086                            header(FCREPO_EVENT_TYPE).contains(DELETE)))
087                    .to("direct:delete.solr")
088                .otherwise()
089                    .to("direct:index.solr");
090
091        /*
092         * Handle re-index events
093         */
094        from("{{solr.reindex.stream}}")
095            .routeId("FcrepoSolrReindex")
096            .to("direct:index.solr");
097
098        /*
099         * Based on an item's metadata, determine if it is indexable.
100         */
101        from("direct:index.solr")
102            .routeId("FcrepoSolrIndexer")
103            .removeHeaders("CamelHttp*")
104            .filter(not(in(tokenizePropertyPlaceholder(getContext(), "{{filter.containers}}", ",").stream()
105                        .map(uri -> or(
106                            header(FCREPO_URI).startsWith(constant(uri + "/")),
107                            header(FCREPO_URI).isEqualTo(constant(uri))))
108                        .collect(toList()))))
109            .choice()
110                .when(and(simple("{{indexing.predicate}} != 'true'"),
111                          simple("{{fcrepo.checkHasIndexingTransformation}} != 'true'")))
112                    .setHeader(INDEXING_TRANSFORMATION).simple("{{fcrepo.defaultTransform}}")
113                    .to("direct:update.solr")
114                .otherwise()
115                    .to("fcrepo:{{fcrepo.baseUrl}}?preferOmit=PreferContainment&accept=application/rdf+xml")
116                    .setHeader(INDEXING_TRANSFORMATION).xpath(hasIndexingTransformation, String.class, ns)
117                    .choice()
118                        .when(or(header(INDEXING_TRANSFORMATION).isNull(),
119                                 header(INDEXING_TRANSFORMATION).isEqualTo("")))
120                            .setHeader(INDEXING_TRANSFORMATION).simple("{{fcrepo.defaultTransform}}").end()
121                    .removeHeaders("CamelHttp*")
122                    .choice()
123                        .when(or(simple("{{indexing.predicate}} != 'true'"),
124                                 header(FCREPO_RESOURCE_TYPE).contains(INDEXABLE)))
125                            .to("direct:update.solr")
126                        .otherwise()
127                            .to("direct:delete.solr");
128
129
130        /*
131         * Remove an item from the solr index.
132         */
133        from("direct:delete.solr").routeId("FcrepoSolrDeleter")
134            .removeHeaders("CamelHttp*")
135            .to("mustache:org/fcrepo/camel/indexing/solr/delete.mustache")
136            .log(LoggingLevel.INFO, logger, "Deleting Solr Object ${headers[CamelFcrepoUri]}")
137            .setHeader(HTTP_METHOD).constant("POST")
138            .setHeader(CONTENT_TYPE).constant("application/json")
139            .setHeader(HTTP_QUERY).simple("commitWithin={{solr.commitWithin}}")
140            .to("{{solr.baseUrl}}/update?useSystemProperties=true");
141
142        from("direct:external.ldpath").routeId("FcrepoSolrLdpathFetch")
143            .removeHeaders("CamelHttp*")
144            .setHeader(HTTP_URI).header(INDEXING_TRANSFORMATION)
145            .setHeader(HTTP_METHOD).constant("GET")
146            .to("http4://localhost/ldpath");
147
148        from("direct:transform.ldpath").routeId("FcrepoSolrTransform")
149            .removeHeaders("CamelHttp*")
150            .setHeader(HTTP_URI).simple("{{ldpath.service.baseUrl}}")
151            .setHeader(HTTP_QUERY).simple("context=${headers.CamelFcrepoUri}")
152            .to("http4://localhost/ldpath");
153
154        /*
155         * Handle update operations
156         */
157        from("direct:update.solr").routeId("FcrepoSolrUpdater")
158            .log(LoggingLevel.INFO, logger, "Indexing Solr Object ${header.CamelFcrepoUri}")
159            .setBody(constant(null))
160            .setHeader(INDEXING_URI).simple("${header.CamelFcrepoUri}")
161            // Don't index the transformation itself
162            .filter().simple("${header.CamelIndexingTransformation} != ${header.CamelIndexingUri}")
163                .choice()
164                    .when(header(INDEXING_TRANSFORMATION).startsWith("http"))
165                        .log(LoggingLevel.INFO, logger,
166                                "Fetching external LDPath program from ${header.CamelIndexingTransformation}")
167                        .to("direct:external.ldpath")
168                        .setHeader(HTTP_METHOD).constant("POST")
169                        .to("direct:transform.ldpath")
170                        .to("direct:send.to.solr")
171                    .when(or(header(INDEXING_TRANSFORMATION).isNull(), header(INDEXING_TRANSFORMATION).isEqualTo("")))
172                        .setHeader(HTTP_METHOD).constant("GET")
173                        .to("direct:transform.ldpath")
174                        .to("direct:send.to.solr")
175                    .otherwise()
176                        .log(LoggingLevel.INFO, logger, "Skipping ${header.CamelFcrepoUri}");
177
178        /*
179         * Send the transformed resource to Solr
180         */
181        from("direct:send.to.solr").routeId("FcrepoSolrSend")
182            .removeHeaders("CamelHttp*")
183            .setHeader(HTTP_METHOD).constant("POST")
184            .setHeader(HTTP_QUERY).simple("commitWithin={{solr.commitWithin}}")
185            .to("{{solr.baseUrl}}/update?useSystemProperties=true");
186    }
187}
