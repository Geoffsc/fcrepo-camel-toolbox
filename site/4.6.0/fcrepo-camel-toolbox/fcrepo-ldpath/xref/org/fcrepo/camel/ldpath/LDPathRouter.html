16  package org.fcrepo.camel.ldpath;
17  
18  import static org.apache.camel.builder.PredicateBuilder.and;
19  import static org.apache.camel.model.dataformat.JsonLibrary.Jackson;
20  import static org.apache.camel.Exchange.CONTENT_TYPE;
21  import static org.apache.camel.Exchange.HTTP_METHOD;
22  import static org.apache.camel.Exchange.HTTP_URI;
23  import static org.slf4j.LoggerFactory.getLogger;
24  
25  import org.apache.camel.builder.RouteBuilder;
26  import org.slf4j.Logger;
27  
28  /**
29   * A content router for an LDPath service.
30   *
31   * @author Aaron Coburn
32   * @since Aug 5, 2016
33   */
34  public class LDPathRouter extends RouteBuilder {
35  
36      private static final Logger LOGGER = getLogger(LDPathRouter.class);
37  
38      public static final String FEDORA_URI = "CamelFedoraUri";
39  
40      /**
41       * Configure the message route workflow.
42       */
43      public void configure() throws Exception {
44  
45          /**
46           * Expose a RESTful endpoint for LDPath processing
47           */
48          from("jetty:http://{{rest.host}}:{{rest.port}}{{rest.prefix}}" +
49                  "?matchOnUriPrefix=true" +
50                  "&httpMethodRestrict=GET,POST,OPTIONS" +
51                  "&sendServerVersion=false")
52              .routeId("FcrepoLDPathRest")
53              .routeDescription("Expose the ldpath endpoint over HTTP")
54              .setHeader(FEDORA_URI).simple("{{fcrepo.baseUrl}}${headers.CamelHttpPath}")
55              .choice()
56                  .when(header(HTTP_METHOD).isEqualTo("GET"))
57                      .to("direct:get")
58                  .when(header(HTTP_METHOD).isEqualTo("POST"))
59                      .to("direct:ldpathPrepare")
60                  .when(header(HTTP_METHOD).isEqualTo("OPTIONS"))
61                      .setHeader(CONTENT_TYPE).constant("text/turtle")
62                      .setHeader("Allow").constant("GET,POST,OPTIONS")
63                      .to("language:simple:resource:classpath:options.ttl");
64  
65          from("direct:get")
66              .routeId("FcrepoLDPathGet")
67              .choice()
68                  .when(and(header("ldpath").isNotNull(), header("ldpath").regex("^https?://.*")))
69                      .setHeader(HTTP_URI).header("ldpath")
70                      .to("http4://localhost?useSystemProperties=true")
71                      .to("direct:ldpathPrepare")
72                  .otherwise()
73                      .to("language:simple:resource:classpath:default.ldpath")
74                      .to("direct:ldpathPrepare");
75  
76          from("direct:ldpathPrepare").routeId("FcrepoLDPathPrepare")
77              .to("direct:ldpath")
78              .to("direct:format");
79  
80          from("direct:format").routeId("FcrepoLDPathFormat")
81              .marshal().json(Jackson)
82              .removeHeaders("*")
83              .setHeader(CONTENT_TYPE).constant("application/json");
84      }
85  }
