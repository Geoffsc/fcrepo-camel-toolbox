16  package org.fcrepo.camel.ldpath;
17  
18  import org.apache.marmotta.commons.http.ContentType;
19  import org.apache.marmotta.ldclient.api.endpoint.Endpoint;
20  
21  /**
22   * @author acoburn
23   * @since Aug 5, 2016
24   */
25  public class FedoraEndpoint extends Endpoint {
26  
27      /**
28       * Create a Fedora endpoint to be used with the Marmotta LDClient
29       * Note: the baseUrl defined here is used by the LDClient to determine
30       * what rules to apply when dereferencing URIs that start with the
31       * given namespace.
32       *
33       * @param namespace the namespace for the Fedora endpoint (e.g. http://localhost:8080/fcrepo/rest/)
34       */
35      public FedoraEndpoint(final String namespace) {
36          this(namespace, 0L);
37      }
38  
39      /**
40       * Create a Fedora endpoint to be used with the Marmotta LDClient
41       * Note: the baseUrl defined here is used by the LDClient to determine
42       * what rules to apply when dereferencing URIs that start with the
43       * given namespace.
44       *
45       * @param namespace the namespace for the Fedora endpoint (e.g. http://localhost:8080/fcrepo/rest/)
46       * @param timeout the length of time (in seconds) to cache the data
47       */
48      public FedoraEndpoint(final String namespace, final Long timeout) {
49          super("Fedora Repository", "Linked Data", namespace + ".*", null, timeout);
50          setPriority(PRIORITY_HIGH);
51          addContentType(new ContentType("application", "rdf+xml", 0.8));
52          addContentType(new ContentType("text", "turtle", 1.0));
53          addContentType(new ContentType("text", "n3", 0.8));
54          addContentType(new ContentType("application", "ld+json", 0.5));
55      }
56  }
