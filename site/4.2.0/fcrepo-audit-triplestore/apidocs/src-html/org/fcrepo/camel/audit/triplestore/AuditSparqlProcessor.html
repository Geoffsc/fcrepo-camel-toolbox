050/**
051 * A processor that converts an audit message into a sparql-update
052 * statement for an external triplestore.
053 *
054 * @author Aaron Coburn
055 * @author escowles
056 * @since 2015-04-09
057 */
058
059public class AuditSparqlProcessor implements Processor {
060
061    /**
062     * Define how a message should be processed.
063     *
064     * @param exchange the current camel message exchange
065     */
066    public void process(final Exchange exchange) throws Exception {
067        final Message in = exchange.getIn();
068        final String eventURIBase = in.getHeader(AuditHeaders.EVENT_BASE_URI, String.class);
069        final String UUIDString = UUID.randomUUID().toString();
070        final UriRef eventURI = new UriRef(eventURIBase + "/" + UUIDString);
071        final Set<Triple> triples = triplesForMessage(in, eventURI);
072
073        // serialize triples
074        final SerializingProvider serializer = new JenaSerializerProvider();
075        final ByteArrayOutputStream serializedGraph = new ByteArrayOutputStream();
076        serializer.serialize(serializedGraph, new SimpleMGraph(triples), "text/rdf+nt");
077
078        // generate SPARQL Update
079        final StringBuilder query = new StringBuilder("update=");
080        query.append(ProcessorUtils.insertData(serializedGraph.toString("UTF-8"), null));
081
082        // update exchange
083        in.setBody(query.toString());
084        in.setHeader(AuditHeaders.EVENT_URI, eventURI.toString());
085        in.setHeader(Exchange.CONTENT_TYPE, "application/x-www-form-urlencoded");
086        in.setHeader(Exchange.HTTP_METHOD, "POST");
087    }
088
089    // namespaces and properties
090    private static final UriRef INTERNAL_EVENT = new UriRef(AUDIT + "InternalEvent");
091    private static final UriRef PREMIS_EVENT = new UriRef(PREMIS + "Event");
092    private static final UriRef PROV_EVENT = new UriRef(PROV + "InstantaneousEvent");
093
094    private static final UriRef PREMIS_TIME = new UriRef(PREMIS + "hasEventDateTime");
095    private static final UriRef PREMIS_OBJ = new UriRef(PREMIS + "hasEventRelatedObject");
096    private static final UriRef PREMIS_AGENT = new UriRef(PREMIS + "hasEventRelatedAgent");
097    private static final UriRef PREMIS_TYPE = new UriRef(PREMIS + "hasEventType");
098
099    private static final UriRef RDF_TYPE = new UriRef(RDF + "type");
100    private static final UriRef XSD_DATE = new UriRef(XSD + "dateTime");
101    private static final UriRef XSD_STRING = new UriRef(XSD + "string");
102
103
104    private static final String EMPTY_STRING = "";
105
106    /**
107     * Convert a Camel message to audit event description.
108     * @param message Camel message produced by an audit event
109     * @param subject RDF subject of the audit description
110     */
111    private static Set<Triple> triplesForMessage(final Message message, final UriRef subject) throws IOException {
112
113        // get info from jms message headers
114        final String eventType = (String) message.getHeader(JmsHeaders.EVENT_TYPE, EMPTY_STRING);
115        final Long timestamp =  (Long) message.getHeader(JmsHeaders.TIMESTAMP, 0);
116        final DateFormat df = new SimpleDateFormat("yyyy-MM-dd'T'HH:mm:ss'Z'");
117        df.setTimeZone(TimeZone.getTimeZone("UTC"));
118        final String date = df.format(new Date(timestamp));
119        final String user = (String) message.getHeader(JmsHeaders.USER, EMPTY_STRING);
120        final String agent = (String) message.getHeader(JmsHeaders.USER_AGENT, EMPTY_STRING);
121        final String properties = (String) message.getHeader(JmsHeaders.PROPERTIES, EMPTY_STRING);
122        final String identifier = ProcessorUtils.getSubjectUri(message);
123        final String premisType = AuditUtils.getAuditEventType(eventType, properties);
124
125        // types
126        final Set<Triple> triples = new HashSet<>();
127        triples.add( new TripleImpl(subject, RDF_TYPE, INTERNAL_EVENT) );
128        triples.add( new TripleImpl(subject, RDF_TYPE, PREMIS_EVENT) );
129        triples.add( new TripleImpl(subject, RDF_TYPE, PROV_EVENT) );
130
131        // basic event info
132        triples.add( new TripleImpl(subject, PREMIS_TIME, new TypedLiteralImpl(date, XSD_DATE)) );
133        triples.add( new TripleImpl(subject, PREMIS_OBJ, new UriRef(identifier)) );
134        triples.add( new TripleImpl(subject, PREMIS_AGENT, new TypedLiteralImpl(user, XSD_STRING)) );
135        triples.add( new TripleImpl(subject, PREMIS_AGENT, new TypedLiteralImpl(agent, XSD_STRING)) );
136        if (premisType != null) {
137            triples.add(new TripleImpl(subject, PREMIS_TYPE, new UriRef(premisType)));
138        }
139        return triples;
140    }
141}
